    <script>
        const socket = io();
        let myId = null;
        let localState = null;

        function joinGame() {
            const name = document.getElementById('playerName').value;
            if (!name) return alert("Por favor, informe seu nome corporativo!");
            socket.emit('join_game', { name: name });
            document.getElementById('login-screen').style.display = 'none';
            document.getElementById('game-screen').style.display = 'flex';
        }

        function startGame() {
            socket.emit('start_game');
        }

        function sendAction(action, amount=0) {
            socket.emit('player_action', { action: action, amount: amount });
        }

        function handleRaiseClick() {
            if (!localState) return;
            
            const p = localState.players[myId];
            const meuFichario = p.chips;
            const apostaMinimaAtual = localState.current_min_bet;
            
            // Regra da Casa: O incremento mínimo é de 5 moedas
            const incrementoMinimo = 5;
            
            // O texto agora pede o valor extra, não o total
            const valorInput = prompt(`🔥 QUANTO VOCÊ QUER ACRESCENTAR?\n\nA aposta atual está em ${apostaMinimaAtual}.\nO incremento mínimo é de ${incrementoMinimo} moedas.\nVocê possui ${meuFichario} moedas na carteira.\n\nDigite APENAS o valor extra que deseja somar à aposta:`);
            
            if (valorInput !== null && valorInput.trim() !== "") {
                const acrescimo = parseInt(valorInput);
                
                if (isNaN(acrescimo)) {
                    alert("❌ Por favor, digite apenas números inteiros!");
                    return;
                }

                // O JavaScript faz a conta por você e descobre qual será a nova aposta total
                const novaApostaTotal = apostaMinimaAtual + acrescimo;
                
                // Verifica se você digitou pelo menos o incremento mínimo de 5 (e se não é um All-in de raspas de tacho)
                if (acrescimo < incrementoMinimo && novaApostaTotal < (meuFichario + p.bet)) {
                    alert(`❌ Aumento Inválido!\nPelas regras da mesa, você precisa acrescentar NO MÍNIMO ${incrementoMinimo} moedas.`);
                } 
                // Verifica se a soma do acréscimo com a aposta atual não ultrapassa o que você tem
                else if (novaApostaTotal > (meuFichario + p.bet)) {
                    alert(`❌ Saldo Insuficiente!\nVocê tentou aumentar em ${acrescimo}, o que daria um total de ${novaApostaTotal}, mas você só tem ${meuFichario + p.bet}. Se quiser apostar tudo, use o botão ALL-IN!`);
                } 
                else {
                    // Envia o valor somado mastigadinho para o Python processar
                    sendAction('raise', novaApostaTotal);
                }
            }
        }

        socket.on('connect', () => { myId = socket.id; });

        socket.on('game_reset', (data) => {
            alert(data.msg);
            location.reload();
        });

        socket.on('show_winner', (data) => {
            alert(data.msg);
        });

        socket.on('error_msg', (data) => {
            alert(data.msg);
            location.reload();
        });

        socket.on('game_update', (state) => {
            localState = state;
            document.getElementById('potAmount').innerText = state.pot;
            
            // Renderizar cartas comunitárias (Mesa)
            const commCardsDiv = document.getElementById('communityCards');
            commCardsDiv.innerHTML = '';
            state.community_cards.forEach(card => {
                const isRed = card.naipe === '♥' || card.naipe === '♦';
                commCardsDiv.innerHTML += `<div class="card ${isRed ? 'red-suit' : ''}"><div>${card.valor}</div><div style="font-size:32px;">${card.naipe}</div></div>`;
            });
            
            const playersContainer = document.getElementById('players-container');
            playersContainer.innerHTML = '';
            
            const playerSids = Object.keys(state.players);
            
            // Exibir o botão Iniciar apenas se estiver aguardando e houver pelo menos 2 na mesa
            if (state.phase === 'waiting' && playerSids.length >= 2) {
                document.getElementById('startBtn').style.display = 'block';
            } else {
                document.getElementById('startBtn').style.display = 'none';
            }

            // --- LÓGICA DA COROA: Encontrar o jogador mais rico (Fichas + Aposta atual) ---
            let maxWealth = -1;
            playerSids.forEach(sid => {
                const wealth = state.players[sid].chips + state.players[sid].bet;
                if (wealth > maxWealth) {
                    maxWealth = wealth;
                }
            });
            
            let posIndex = 0;
            playerSids.forEach((sid) => {
                const p = state.players[sid];
                const seat = document.createElement('div');
                
                seat.className = `player-seat pos-${sid === myId ? 0 : posIndex + 1}`;
                
                let statusInfo = "";
                if (p.status === 'folded') statusInfo = "<span style='color:red;'>Folded</span>";
                else if (p.status === 'all-in') statusInfo = "<span style='color:orange;'>ALL-IN</span>";
                else if (p.bet > 0) statusInfo = `<span class="bet-tag">Apostou: ${p.bet}</span>`;

                // --- DESENHANDO A COROA ---
                // Agora ela tem left, translateX e z-index altíssimo para centralizar e não sumir!
                const myWealth = p.chips + p.bet;
                let crownHtml = "";
                if (myWealth === maxWealth && maxWealth > 0) {
                    crownHtml = "<div style='position: absolute; top: -35px; left: 50%; transform: translateX(-50%); font-size: 32px; z-index: 999; text-shadow: 0px 0px 10px gold, 0px 4px 4px rgba(0,0,0,0.8);'>👑</div>";
                }
                
                seat.innerHTML = `
                    ${crownHtml}
                    <span class="name">${p.name}</span>
                    <span class="chips">${p.chips} 💰</span>
                    ${statusInfo}
                `;
                
                // Brilho amarelo/dourado se for o turno da pessoa
                if (state.phase !== 'waiting' && state.turn_order[state.current_turn_index] === sid) {
                    seat.style.borderColor = '#f2c94c';
                    seat.style.boxShadow = '0 0 25px #f2c94c';
                    seat.style.transform = sid === myId ? 'translateX(-50%) scale(1.1)' : 'scale(1.1)';
                }

                playersContainer.appendChild(seat);
                
                // Trata elementos do jogador dono da tela
                if (sid === myId) {
                    const myCardsDiv = document.getElementById('myCards');
                    myCardsDiv.innerHTML = '';
                    p.cards.forEach(card => {
                        const isRed = card.naipe === '♥' || card.naipe === '♦';
                        myCardsDiv.innerHTML += `<div class="card ${isRed ? 'red-suit' : ''}"><div>${card.valor}</div><div style="font-size:32px;">${card.naipe}</div></div>`;
                    });

                    const btnCall = document.getElementById('callBtn');
                    if (btnCall) {
                        const betNeeded = state.current_min_bet - p.bet;
                        if (betNeeded > 0) {
                            btnCall.innerText = `Pagar (Call ${betNeeded})`;
                        } else {
                            btnCall.innerText = "Pedir Mesa (Check)";
                        }
                    }

                    const isMyTurn = state.turn_order[state.current_turn_index] === myId;
                    document.getElementById('actionPanel').style.display = (isMyTurn && state.phase !== 'waiting') ? 'flex' : 'none';
                }
                
                if (sid !== myId) posIndex++;
            });
        });
    </script>
