    <script>
        const socket = io(); let myId = null; let myName = ""; 
        let localPoker = null; let localPifPaf = null; let pifpafSelectedCard = -1; let myPifPafCards = [];
        
        let draggedIdx = null; 
        let touchIdx = null;

        socket.on('connect', () => { myId = socket.id; });
        socket.on('game_reset', (data) => { alert(data.msg); location.reload(); });

        // Notificação de Ações (Aumento e All-in)
        socket.on('action_notification', (data) => {
            const toast = document.getElementById('toast-msg');
            toast.innerText = data.msg;
            toast.style.background = '#2196F3'; // Azul no Poker
            toast.style.boxShadow = '0 5px 25px rgba(33, 150, 243, 0.8)';
            toast.style.display = 'block';
            
            setTimeout(() => { 
                toast.style.display = 'none'; 
                // Devolve as cores originais (vermelho de erro) quando fechar
                toast.style.background = '#ef4444'; 
                toast.style.boxShadow = '0 5px 15px rgba(0,0,0,0.5)';
            }, 4000);
        });
        
        // PLACA DE ERRO BONITA E VISÍVEL
        socket.on('error_msg', (data) => { 
            const toast = document.getElementById('toast-msg');
            toast.innerText = data.msg;
            toast.style.display = 'block';
            setTimeout(() => { toast.style.display = 'none'; }, 4000); 
        });

        function goToLobby() {
            const nameInput = document.getElementById('playerName').value;
            if (!nameInput) return alert("Insira um nome válido!");
            myName = nameInput; document.getElementById('login-screen').style.display = 'none'; document.getElementById('lobby-screen').style.display = 'block';
        }
        function selectGame(gameType) {
            socket.emit('join_game', { name: myName, game: gameType });
            document.getElementById('lobby-screen').style.display = 'none';
            if(gameType === 'poker') document.getElementById('poker-screen').style.display = 'flex';
            if(gameType === 'pifpaf') document.getElementById('pifpaf-screen').style.display = 'flex';
        }
        function getCardHTML(card) {
            const isRed = card.naipe === '♥' || card.naipe === '♦';
            return `<div class="card ${isRed ? 'red-suit' : ''}"><div>${card.valor}</div><div style="font-size:32px;">${card.naipe}</div></div>`;
        }

        // =====================================
        // LOGICA PIF PAF E DRAG AND DROP
        // =====================================
        function selectPifPafCard(idx) { pifpafSelectedCard = idx; renderPifPafHand(); }
        
        function dragStart(e, idx) { draggedIdx = idx; e.dataTransfer.effectAllowed = 'move'; }
        function dragOver(e) { e.preventDefault(); e.dataTransfer.dropEffect = 'move'; }
        function drop(e, targetIdx) {
            e.preventDefault();
            if (draggedIdx === null || draggedIdx === targetIdx) return;
            
            const draggedElement = document.querySelector(`[data-idx="${draggedIdx}"]`);
            draggedElement.style.opacity = '0';
            
            setTimeout(() => {
                const card = myPifPafCards.splice(draggedIdx, 1)[0];
                myPifPafCards.splice(targetIdx, 0, card);
                
                if (pifpafSelectedCard === draggedIdx) pifpafSelectedCard = targetIdx;
                else if (draggedIdx < pifpafSelectedCard && targetIdx >= pifpafSelectedCard) pifpafSelectedCard--;
                else if (draggedIdx > pifpafSelectedCard && targetIdx <= pifpafSelectedCard) pifpafSelectedCard++;
                
                draggedIdx = null; 
                renderPifPafHand();
            }, 150); 
        }

        function touchStart(e, idx) { touchIdx = idx; e.currentTarget.style.transform = 'translateY(-10px)'; }
        function touchMove(e) { if (touchIdx !== null) { e.preventDefault(); } }
        function touchEnd(e) {
            if (touchIdx === null) return;
            const touch = e.changedTouches[0];
            const targetEl = document.elementFromPoint(touch.clientX, touch.clientY);
            
            if (targetEl) {
                const dropCard = targetEl.closest('.my-pifpaf-hand .card');
                if (dropCard && dropCard.hasAttribute('data-idx')) {
                    const targetIdx = parseInt(dropCard.getAttribute('data-idx'));
                    if (targetIdx !== touchIdx) {
                        const card = myPifPafCards.splice(touchIdx, 1)[0];
                        myPifPafCards.splice(targetIdx, 0, card);
                        
                        if (pifpafSelectedCard === touchIdx) pifpafSelectedCard = targetIdx;
                        else if (touchIdx < pifpafSelectedCard && targetIdx >= pifpafSelectedCard) pifpafSelectedCard--;
                        else if (touchIdx > pifpafSelectedCard && targetIdx <= pifpafSelectedCard) pifpafSelectedCard++;
                    }
                }
            }
            touchIdx = null; renderPifPafHand();
        }

        function renderPifPafHand() {
            const div = document.getElementById('pifpaf-myCards'); 
            div.innerHTML = '';
            
            myPifPafCards.forEach((c, i) => {
                const isRed = c.naipe === '♥' || c.naipe === '♦';
                const selClass = i === pifpafSelectedCard ? 'selected' : '';
                
                // Criamos o elemento da carta
                const cardEl = document.createElement('div');
                cardEl.className = `card ${isRed ? 'red-suit' : ''} ${selClass}`;
                cardEl.innerHTML = `<div>${c.valor}</div><div style="font-size:32px;">${c.naipe}</div>`;
                cardEl.setAttribute('data-idx', i);
                cardEl.setAttribute('draggable', 'true');
                
                // Adicionamos os eventos com um pequeno filtro para não conflitar
                cardEl.addEventListener('dragstart', (e) => dragStart(e, i));
                cardEl.addEventListener('dragover', dragOver);
                cardEl.addEventListener('drop', (e) => drop(e, i));
                
                // TRUQUE: O clique agora usa 'mousedown' para ser mais rápido e preciso no PC
                cardEl.addEventListener('mousedown', (e) => {
                    // Impede que o clique dispare se for apenas um arrasto iniciado
                    selectPifPafCard(i);
                });
                
                // Mantém os eventos de touch para o celular
                cardEl.addEventListener('touchstart', (e) => touchStart(e, i));
                cardEl.addEventListener('touchmove', touchMove);
                cardEl.addEventListener('touchend', touchEnd);
                
                div.appendChild(cardEl);
            });
            
            // ... (O restante da lógica de botões e visibilidade continua igual)
            if(!localPifPaf) return;
            const isMyTurn = localPifPaf.turn_order[localPifPaf.current_turn_index] === myId;
            const isPlaying = localPifPaf.phase === 'playing';
            const has9 = myPifPafCards.length === 9; 
            const has10 = myPifPafCards.length === 10;
            
            document.getElementById('pifpaf-actionPanel').style.display = isPlaying ? 'flex' : 'none';
            document.getElementById('btn-discard').style.display = 'none';
            document.getElementById('btn-bater').style.display = 'none';
            document.getElementById('btn-intercept').style.display = 'none';

            if(isMyTurn && isPlaying) {
                document.getElementById('btn-discard').style.display = (has10 && pifpafSelectedCard !== -1) ? 'block' : 'none';
                document.getElementById('btn-bater').style.display = (has10 && pifpafSelectedCard !== -1) ? 'block' : 'none';
            } else if (isPlaying && has9 && pifpafSelectedCard !== -1 && localPifPaf.lixo.length > 0) {
                document.getElementById('btn-intercept').style.display = 'block';
            }
        }

        function drawMonte() { if(myPifPafCards.length === 9) socket.emit('pifpaf_action', {action: 'draw_monte'}); }
        function drawLixo() { if(myPifPafCards.length === 9) socket.emit('pifpaf_action', {action: 'draw_lixo'}); }
        
        function discardCard() {
            if (pifpafSelectedCard !== -1) {
                const card = myPifPafCards[pifpafSelectedCard];
                socket.emit('pifpaf_action', {action: 'discard', card_id: card.id});
                pifpafSelectedCard = -1;
            }
        }

        function baterPifPaf() {
            if (pifpafSelectedCard !== -1) {
                if (confirm("Deseja BATER descartando esta carta? O servidor vai analisar se suas 9 cartas formam jogos válidos.")) {
                    const card = myPifPafCards[pifpafSelectedCard];
                    socket.emit('pifpaf_action', { action: 'bater', card_id: card.id });
                    pifpafSelectedCard = -1;
                }
            }
        }

        function interceptBater() {
            if (pifpafSelectedCard !== -1) {
                if (confirm("⚡ INTERCEPTAÇÃO: Tem certeza que a carta que está no LIXO completa seu jogo? Você descartará a carta selecionada.")) {
                    const card = myPifPafCards[pifpafSelectedCard];
                    socket.emit('pifpaf_action', { action: 'intercept_bater', card_id: card.id });
                    pifpafSelectedCard = -1;
                }
            }
        }

        socket.on('pifpaf_update', (state) => {
            localPifPaf = state; const pids = Object.keys(state.players);
            document.getElementById('startPifPafBtn').style.display = (state.phase === 'waiting' && pids.length >= 2) ? 'block' : 'none';
            const lixoDiv = document.getElementById('lixo-pile');
            if(state.lixo.length > 0) {
                const topo = state.lixo[state.lixo.length - 1];
                lixoDiv.innerHTML = getCardHTML(topo).replace('class="card', 'class="'); lixoDiv.className = topo.naipe === '♥' || topo.naipe === '♦' ? 'card red-suit' : 'card';
            } else { lixoDiv.innerHTML = ''; lixoDiv.className = 'card'; }

            const playersDiv = document.getElementById('pifpaf-players'); playersDiv.innerHTML = '';
            let posIndex = 0;

                    // --- SINCRONIZAÇÃO INTELIGENTE BLINDADA ---
                    let serverCards = p.cards;
                    
                    // Se for nova rodada ou reconnect (Diferença brusca de cartas)
                    if (myPifPafCards.length === 0 || serverCards.length === 0 || Math.abs(serverCards.length - myPifPafCards.length) > 1) {
                        myPifPafCards = [...serverCards];
                    } else {
                        // Se comprou (Adiciona na direita)
                        if (serverCards.length > myPifPafCards.length) {
                            let novas = serverCards.filter(sc => !myPifPafCards.find(lc => lc.id === sc.id));
                            myPifPafCards.push(...novas); 
                        }
                        // Se descartou (Remove apenas o que saiu do servidor)
                        else if (serverCards.length < myPifPafCards.length) {
                            myPifPafCards = myPifPafCards.filter(lc => serverCards.find(sc => sc.id === lc.id));
                        }
                    }
                    renderPifPafHand();
                }
                const seat = document.createElement('div'); seat.className = `player-seat pos-${sid === myId ? 0 : posIndex + 1}`;
                seat.innerHTML = `<span class="name">${p.name}</span><span class="chips">${p.cards.length} Cartas 🎴</span>`;

        const socket = io(); let myId = null; let myName = "";
        let localPoker = null; let localPifPaf = null; let localDomino = null; 
        
        // PifPaf Vars
        let pifpafSelectedCard = -1; let myPifPafCards = []; let draggedIdx = null; let touchIdx = null;
        // Domino Vars
        let draggedDominoId = null;

        socket.on('connect', () => { myId = socket.id; });

        socket.on('action_notification', (data) => { showToast(data.msg, '#2196F3'); });
        
        function showToast(msg, color) {
            const toast = document.getElementById('toast-msg'); toast.innerText = msg; toast.style.background = color; toast.style.display = 'block';
            setTimeout(() => { toast.style.display = 'none'; toast.style.background = '#ef4444'; }, 4000);
        }

        function goToLobby() {
            myName = document.getElementById('playerName').value;

        function selectGame(g) {

            document.getElementById('lobby-screen').style.display = 'none';
            document.getElementById(g+'-screen').style.display = 'flex';

        function getPips(n) {
            const grid = Array(9).fill(0);
            if (n === 1) { grid[4] = 1; }
            else if (n === 2) { grid[2] = 1; grid[6] = 1; }
            else if (n === 3) { grid[2] = 1; grid[4] = 1; grid[6] = 1; }
            else if (n === 4) { grid[0] = 1; grid[2] = 1; grid[6] = 1; grid[8] = 1; }
            else if (n === 5) { grid[0] = 1; grid[2] = 1; grid[4] = 1; grid[6] = 1; grid[8] = 1; }
            else if (n === 6) { grid[0] = 1; grid[2] = 1; grid[3] = 1; grid[5] = 1; grid[6] = 1; grid[8] = 1; }
            
            let html = '<div class="face">';
            for(let i = 0; i < 9; i++) {
                html += `<div class="pip" style="opacity:${grid[i]}"></div>`;
            }
            html += '</div>';
            return html;
        }

        function getDominoHTML(t, isHand = false) {
            // Só deita a peça dupla (carroção) se ela estiver NA MESA. Na mão fica em pé.
            const isDouble = (!isHand && t.left === t.right) ? 'double' : '';
            return `<div class="domino-tile ${isDouble}">${getPips(t.left)}<div class="divider"></div>${getPips(t.right)}</div>`;
        }

        ['left', 'right'].forEach(side => {
            const zone = document.getElementById(`drop-${side}`);
            zone.addEventListener('dragover', e => { e.preventDefault(); zone.classList.add('dragover'); });
            zone.addEventListener('dragleave', e => { zone.classList.remove('dragover'); });
            zone.addEventListener('drop', e => {
                e.preventDefault(); zone.classList.remove('dragover');
                if(draggedDominoId) socket.emit('domino_action', {action: 'play', tile_id: draggedDominoId, side: side});
            });
        });

        socket.on('domino_update', (state) => {
            localDomino = state; const pids = Object.keys(state.players);
            document.getElementById('startDominoBtn').style.display = (state.phase === 'waiting' && pids.length >= 2) ? 'block' : 'none';
            
            const chain = document.getElementById('domino-chain'); chain.innerHTML = '';
            state.board.forEach(t => { chain.innerHTML += getDominoHTML(t, false); });

            const playersDiv = document.getElementById('domino-players'); playersDiv.innerHTML = '';
            let posIdx = 0;
            pids.forEach(sid => {
                const p = state.players[sid];
                if(sid === myId) {
                    const myHand = document.getElementById('domino-myCards'); myHand.innerHTML = '';
                    p.cards.forEach(t => {
                        const tDiv = document.createElement('div'); tDiv.innerHTML = getDominoHTML(t, true); const el = tDiv.firstChild;
                        el.setAttribute('draggable', 'true');
                        el.addEventListener('dragstart', (e) => { draggedDominoId = t.id; });
                        el.addEventListener('touchstart', (e) => { draggedDominoId = t.id; el.style.transform = 'translateY(-20px)'; });
                        el.addEventListener('touchmove', (e) => { e.preventDefault(); });
                        el.addEventListener('touchend', (e) => {
                            const touch = e.changedTouches[0]; const dropEl = document.elementFromPoint(touch.clientX, touch.clientY);
                            if(dropEl) {
                                if(dropEl.closest('#drop-left')) socket.emit('domino_action', {action: 'play', tile_id: draggedDominoId, side: 'left'});
                                if(dropEl.closest('#drop-right')) socket.emit('domino_action', {action: 'play', tile_id: draggedDominoId, side: 'right'});
                            }
                            el.style.transform = ''; draggedDominoId = null;
                        });
                        myHand.appendChild(el);
                    });
                }
                const seat = document.createElement('div'); seat.className = `player-seat pos-${sid === myId ? 0 : posIdx + 1}`;
                seat.innerHTML = `<span class="name">${p.name}</span><span style="color:#f59e0b; font-weight:bold;">${p.cards.length} Peças 🎲</span>`;
                if(state.phase === 'playing' && state.turn_order[state.current_turn_index] === sid) {
                    seat.style.border = '3px solid #f59e0b'; seat.style.animation = 'pulse-domino 1.2s infinite';
                    seat.style.transform = sid === myId ? 'translateX(-50%) scale(1.15)' : 'scale(1.15)'; seat.style.zIndex = '50';
                }
                playersDiv.appendChild(seat);
                if(sid !== myId) posIdx++;
            });
            const isMyTurn = state.phase === 'playing' && state.turn_order[state.current_turn_index] === myId;
            document.getElementById('domino-actionPanel').style.display = isMyTurn ? 'flex' : 'none';
        });

        function selectPifPafCard(idx) { pifpafSelectedCard = idx; renderPifPafHand(); }
        function dragStart(e, idx) { draggedIdx = idx; e.dataTransfer.effectAllowed = 'move'; }
        function dragOver(e) { e.preventDefault(); e.dataTransfer.dropEffect = 'move'; }
        function drop(e, targetIdx) {
            e.preventDefault(); if (draggedIdx === null || draggedIdx === targetIdx) return;
            const draggedElement = document.querySelector(`[data-idx="${draggedIdx}"]`); draggedElement.style.opacity = '0';
            setTimeout(() => {
                const card = myPifPafCards.splice(draggedIdx, 1)[0]; myPifPafCards.splice(targetIdx, 0, card);
                if (pifpafSelectedCard === draggedIdx) pifpafSelectedCard = targetIdx;
                else if (draggedIdx < pifpafSelectedCard && targetIdx >= pifpafSelectedCard) pifpafSelectedCard--;
                else if (draggedIdx > pifpafSelectedCard && targetIdx <= pifpafSelectedCard) pifpafSelectedCard++;
                draggedIdx = null; renderPifPafHand();
            }, 150); 
        }
        function touchStart(e, idx) { touchIdx = idx; e.currentTarget.style.transform = 'translateY(-10px)'; }
        function touchMove(e) { if (touchIdx !== null) { e.preventDefault(); } }
        function touchEnd(e) {
            if (touchIdx === null) return;
            const touch = e.changedTouches[0]; const targetEl = document.elementFromPoint(touch.clientX, touch.clientY);
            if (targetEl) {
                const dropCard = targetEl.closest('.my-pifpaf-hand .card');
                if (dropCard && dropCard.hasAttribute('data-idx')) {
                    const targetIdx = parseInt(dropCard.getAttribute('data-idx'));
                    if (targetIdx !== touchIdx) {
                        const card = myPifPafCards.splice(touchIdx, 1)[0]; myPifPafCards.splice(targetIdx, 0, card);
                        if (pifpafSelectedCard === touchIdx) pifpafSelectedCard = targetIdx;
                        else if (touchIdx < pifpafSelectedCard && targetIdx >= pifpafSelectedCard) pifpafSelectedCard--;
                        else if (touchIdx > pifpafSelectedCard && targetIdx <= pifpafSelectedCard) pifpafSelectedCard++;
                    }
                }
            }
            touchIdx = null; renderPifPafHand();
        }

        function renderPifPafHand() {
            const div = document.getElementById('pifpaf-myCards'); div.innerHTML = '';
            myPifPafCards.forEach((c, i) => {
                const isRed = c.naipe === '♥' || c.naipe === '♦';
                const selClass = i === pifpafSelectedCard ? 'selected' : '';
                const cardEl = document.createElement('div'); cardEl.className = `card ${isRed ? 'red-suit' : ''} ${selClass}`;
                cardEl.innerHTML = `<div>${c.valor}</div><div style="font-size:32px;">${c.naipe}</div>`;
                cardEl.setAttribute('data-idx', i); cardEl.setAttribute('draggable', 'true');
                cardEl.addEventListener('dragstart', (e) => dragStart(e, i)); cardEl.addEventListener('dragover', dragOver); cardEl.addEventListener('drop', (e) => drop(e, i));
                cardEl.addEventListener('mousedown', (e) => { selectPifPafCard(i); });
                cardEl.addEventListener('touchstart', (e) => touchStart(e, i)); cardEl.addEventListener('touchmove', touchMove); cardEl.addEventListener('touchend', touchEnd);
                div.appendChild(cardEl);
            });
            if(!localPifPaf) return;
            const isMyTurn = localPifPaf.turn_order[localPifPaf.current_turn_index] === myId;
            const isPlaying = localPifPaf.phase === 'playing';
            const has9 = myPifPafCards.length === 9; const has10 = myPifPafCards.length === 10;
            
            document.getElementById('pifpaf-actionPanel').style.display = isPlaying ? 'flex' : 'none';
            document.getElementById('btn-discard').style.display = 'none'; document.getElementById('btn-bater').style.display = 'none'; document.getElementById('btn-intercept').style.display = 'none';
            if(isMyTurn && isPlaying) {
                document.getElementById('btn-discard').style.display = (has10 && pifpafSelectedCard !== -1) ? 'block' : 'none';
                document.getElementById('btn-bater').style.display = (has10 && pifpafSelectedCard !== -1) ? 'block' : 'none';
                document.getElementById('monte-pile').style.boxShadow = (!has10) ? '0 0 20px #10b981' : ''; document.getElementById('lixo-pile').style.boxShadow = (!has10) ? '0 0 20px #ef4444' : '';
            } else if (isPlaying && has9 && pifpafSelectedCard !== -1 && localPifPaf.lixo.length > 0) {
                document.getElementById('btn-intercept').style.display = 'block'; document.getElementById('monte-pile').style.boxShadow = ''; document.getElementById('lixo-pile').style.boxShadow = '';
            }
        }

        function drawMonte() { if(myPifPafCards.length === 9) socket.emit('pifpaf_action', {action: 'draw_monte'}); }
        function drawLixo() { if(myPifPafCards.length === 9) socket.emit('pifpaf_action', {action: 'draw_lixo'}); }
        function discardCard() { if (pifpafSelectedCard !== -1) { socket.emit('pifpaf_action', {action: 'discard', card_id: myPifPafCards[pifpafSelectedCard].id}); pifpafSelectedCard = -1; } }
        function baterPifPaf() { if (pifpafSelectedCard !== -1 && confirm("Bater descartando esta carta?")) { socket.emit('pifpaf_action', { action: 'bater', card_id: myPifPafCards[pifpafSelectedCard].id }); pifpafSelectedCard = -1; } }
        function interceptBater() { if (pifpafSelectedCard !== -1 && confirm("Roubar lixo e bater?")) { socket.emit('pifpaf_action', { action: 'intercept_bater', card_id: myPifPafCards[pifpafSelectedCard].id }); pifpafSelectedCard = -1; } }

        socket.on('pifpaf_update', (state) => {
            localPifPaf = state; const pids = Object.keys(state.players);
            document.getElementById('startPifPafBtn').style.display = (state.phase === 'waiting' && pids.length >= 2) ? 'block' : 'none';
            const lixoDiv = document.getElementById('lixo-pile');
            if(state.lixo.length > 0) {
                const topo = state.lixo[state.lixo.length - 1]; const isRed = topo.naipe === '♥' || topo.naipe === '♦';
                lixoDiv.innerHTML = `<div>${topo.valor}</div><div style="font-size:32px;">${topo.naipe}</div>`; lixoDiv.className = isRed ? 'card red-suit' : 'card';
            } else { lixoDiv.innerHTML = ''; lixoDiv.className = 'card'; }

            const playersDiv = document.getElementById('pifpaf-players'); playersDiv.innerHTML = '';
            let posIndex = 0;
            pids.forEach((sid) => {
                const p = state.players[sid];
                if(sid === myId) {
                    let serverCards = p.cards;
                    if (myPifPafCards.length === 0 || serverCards.length === 0 || Math.abs(serverCards.length - myPifPafCards.length) > 1) { myPifPafCards = [...serverCards]; }
                    else {
                        if (serverCards.length > myPifPafCards.length) { let novas = serverCards.filter(sc => !myPifPafCards.find(lc => lc.id === sc.id)); myPifPafCards.push(...novas); }
                        else if (serverCards.length < myPifPafCards.length) { myPifPafCards = myPifPafCards.filter(lc => serverCards.find(sc => sc.id === lc.id)); }
                    }
                    renderPifPafHand();
                }
                const seat = document.createElement('div'); seat.className = `player-seat pos-${sid === myId ? 0 : posIndex + 1}`;
                seat.innerHTML = `<span class="name">${p.name}</span><span class="chips">${p.cards.length} Cartas 🎴</span>`;
                if (state.phase !== 'waiting' && state.turn_order[state.current_turn_index] === sid) { seat.style.border = '3px solid #10b981'; seat.style.animation = 'pulse-pifpaf 1.2s infinite'; seat.style.transform = sid === myId ? 'translateX(-50%) scale(1.15)' : 'scale(1.15)'; seat.style.zIndex = '50'; }
                playersDiv.appendChild(seat);
                if(sid !== myId) posIndex++;
            });
            renderPifPafHand();
        });

        function sendPokerAction(act, amt=0) { socket.emit('player_action', { action: act, amount: amt }); }
        function handleRaiseClick() {
            if (!localPoker) return; const p = localPoker.players[myId]; const min = localPoker.current_min_bet;
            const val = prompt(`Aposta atual: ${min}\nFichas: ${p.chips}\nAumentar em quanto?`);
            if (val !== null && val.trim() !== "") {
                const num = parseInt(val); if (isNaN(num)) return alert("Digite números inteiros.");
                const nova = min + num;
                if (num < 5 && nova < (p.chips + p.bet)) alert(`Aumente no mínimo 5.`); else if (nova > (p.chips + p.bet)) alert(`Saldo Insuficiente.`); else sendPokerAction('raise', nova);
            }
        }
        socket.on('showdown', (data) => {
            const modal = document.getElementById('showdown-modal'); document.getElementById('showdown-msg').innerText = data.msg;
            const handsContainer = document.getElementById('showdown-hands'); handsContainer.innerHTML = '';
            data.hands.forEach(h => {
                let cardsHtml = ''; h.cards.forEach(c => { const isRed = c.naipe === '♥' || c.naipe === '♦'; cardsHtml += `<div class="card ${isRed?'red-suit':''}" style="width:65px; height:95px; font-size:22px;"><div>${c.valor}</div><div style="font-size:32px;">${c.naipe}</div></div>`; });
                const ws = h.is_winner ? 'border: 2px solid #2196F3; opacity: 1;' : 'border: 1px solid #334155; opacity: 0.6;';
                const hl = h.hand_name ? `<div style="color:#64b5f6; font-size:13px;">[ ${h.hand_name} ]</div>` : '';
                handsContainer.innerHTML += `<div style="text-align:center; padding:15px; border-radius:10px; ${ws}"><h3 style="margin-top:0;">${h.is_winner?'🏆 ':''}${h.name}</h3>${hl}<div style="display:flex; gap:8px;">${cardsHtml}</div></div>`;
            });
            modal.style.display = 'flex'; setTimeout(() => { modal.style.display = 'none'; }, 5000);
        });
        socket.on('game_update', (state) => {
            localPoker = state; document.getElementById('potAmount').innerText = state.pot;
            const commCardsDiv = document.getElementById('communityCards'); commCardsDiv.innerHTML = '';
            state.community_cards.forEach(c => { const isRed = c.naipe === '♥' || c.naipe === '♦'; commCardsDiv.innerHTML += `<div class="card ${isRed?'red-suit':''}"><div>${c.valor}</div><div style="font-size:32px;">${c.naipe}</div></div>`; });
            const pids = Object.keys(state.players);
            document.getElementById('startBtn').style.display = (state.phase === 'waiting' && pids.length >= 2) ? 'block' : 'none';
            let maxW = -1; pids.forEach(sid => { const w = state.players[sid].chips + state.players[sid].bet; if (w > maxW) maxW = w; });
            const playersDiv = document.getElementById('poker-players'); playersDiv.innerHTML = '';
            let posIndex = 0;
            pids.forEach((sid) => {
                const p = state.players[sid]; const seat = document.createElement('div'); seat.className = `player-seat pos-${sid === myId ? 0 : posIndex + 1}`;
                let sInfo = ""; if (p.status === 'folded') sInfo = "<span style='color:#ef4444;'>Folded</span>"; else if (p.status === 'all-in') sInfo = "<span style='color:#f59e0b;'>ALL-IN</span>"; else if (p.bet > 0) sInfo = `<span style="color:#94a3b8;">Apostou: ${p.bet}</span>`;
                const mw = p.chips + p.bet; let cHtml = (mw === maxW && maxW > 0) ? "<div style='position:absolute; top:-35px; left:50%; transform:translateX(-50%); font-size:32px; z-index:999;'>👑</div>" : "";
                seat.innerHTML = `${cHtml}<span class="name">${p.name}</span><span class="chips">${p.chips} 💻</span>${sInfo}`;
                if (state.phase !== 'waiting' && state.turn_order[state.current_turn_index] === sid) { seat.style.border = '3px solid #2196F3'; seat.style.animation = 'pulse-poker 1.2s infinite'; seat.style.transform = sid === myId ? 'translateX(-50%) scale(1.15)' : 'scale(1.15)'; seat.style.zIndex = '50'; }
                playersDiv.appendChild(seat);
                if (sid === myId) {
                    const myCardsDiv = document.getElementById('poker-myCards'); myCardsDiv.innerHTML = ''; p.cards.forEach(c => { const isRed = c.naipe === '♥' || c.naipe === '♦'; myCardsDiv.innerHTML += `<div class="card ${isRed?'red-suit':''}"><div>${c.valor}</div><div style="font-size:32px;">${c.naipe}</div></div>`; });
                    const btnCall = document.getElementById('callBtn'); if (btnCall) { const bn = state.current_min_bet - p.bet; btnCall.innerText = bn > 0 ? `Pagar (${bn})` : "Pedir Mesa (Check)"; }
                    document.getElementById('poker-actionPanel').style.display = (state.turn_order[state.current_turn_index] === myId && state.phase !== 'waiting') ? 'flex' : 'none';
                }
                if (sid !== myId) posIndex++;
            });
        });
