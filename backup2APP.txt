# Estrutura: {sid: {chips, bet, status, cards, current_round_bet}}
game_state = {
    'players': {},
    'pot': 0,
    'side_pots': [], # Lista de dicionários: {'amount': int, 'eligible_players': []}
    'community_cards': [],
    'phase': 'waiting',
    'current_min_bet': 50,
    'last_raise': 50,
    'dealer_index': 0,
    'turn_order': [],
    'current_turn_index': 0
}

def process_all_in(sid):
    p = game_state['players'][sid]
    all_in_amt = p['chips'] + p['bet']
    p['bet'] = all_in_amt
    p['chips'] = 0
    p['status'] = 'all-in'
    
    # Lógica de criação de Side Pot se o valor for menor que a aposta atual
    if p['bet'] < game_state['current_min_bet']:
        # O jogador criou um pote paralelo
        pass 
    elif p['bet'] > game_state['current_min_bet']:
        game_state['last_raise'] = p['bet'] - game_state['current_min_bet']
        game_state['current_min_bet'] = p['bet']

@socketio.on('player_action')
def handle_action(data):
    sid = request.sid
    action = data.get('action')
    amount = int(data.get('amount', 0))
    p = game_state['players'][sid]

    # Validação de Segurança (Cenário 17)
    if sid != game_state['turn_order'][game_state['current_turn_index']]: return

    if action == 'fold':
        p['status'] = 'folded'
    
    elif action == 'call':
        needed = game_state['current_min_bet'] - p['bet']
        if p['chips'] <= needed: # Cenário 18: All-in forçado
            p['bet'] += p['chips']
            p['chips'] = 0
            p['status'] = 'all-in'
        else:
            p['chips'] -= needed
            p['bet'] += needed

    elif action == 'raise':
        # Cenário 19: Validação de Raise Mínimo
        if amount < (game_state['current_min_bet'] + game_state['last_raise']):
            emit('error_msg', {'msg': 'Raise muito baixo!'})
            return
        
        diff = amount - p['bet']
        if diff > p['chips']: return # Sem fundos
        p['chips'] -= diff
        p['bet'] = amount
        game_state['last_raise'] = amount - game_state['current_min_bet']
        game_state['current_min_bet'] = amount

    # ... (restante da lógica de transição de fase com verificação de todos igualados)
    emit('game_update', game_state, broadcast=True)
